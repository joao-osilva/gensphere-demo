command to generate cert:

sudo certbot --nginx -d gensphere.io -d registry.gensphere.io

nginx conf at /etc/nginx/sites-available/default


server {
    listen 80;
    listen [::]:80;
    server_name gensphere.io registry.gensphere.io;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    listen [::]:443 ssl;
    server_name gensphere.io;

    ssl_certificate /etc/letsencrypt/live/gensphere.io/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/gensphere.io/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    location / {
        proxy_pass http://localhost:8501;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
    }
}
